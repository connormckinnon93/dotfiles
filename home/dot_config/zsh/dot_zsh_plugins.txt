#!/bin/zsh
#
# .zsh_plugins.txt - antidote plugin list. Docs: https://antidote.sh/
#

# Completion system - ez-compinit manages compinit timing and caching.
mattmc3/ez-compinit

# Fuzzy Tab-completion UI. Needs the fzf binary; sits on top of carapace's
# completions (carapace itself is initialized in .zshrc.d/carapace.zsh).
aloxaf/fzf-tab

# Replace cat with bat (syntax highlighting) and colorize man pages. Needs the
# bat binary (chezmoiexternal); use `rcat` for the real cat.
fdellwing/zsh-bat

# Inline "as you type" suggestions (config in .zstyles). Loads after fzf-tab
# and before syntax highlighting.
zsh-users/zsh-autosuggestions

# Syntax highlighting. Must be sourced after autosuggestions and before
# history-substring-search.
zdharma-continuum/fast-syntax-highlighting

# History substring search (prefix + Up/Down). Must be sourced last, after
# syntax highlighting. Keybindings live in .zshrc.d/history-substring-search.zsh.
zsh-users/zsh-history-substring-search
